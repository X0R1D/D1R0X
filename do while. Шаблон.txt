void main()
{

	int pred=0, nin=0;
	int i=0;

	do {
		cout << "введите 0 или 1" << endl;
		cin >> nin;

		if (nin == pred && pred == 1) {
			i++;
		}

		pred = nin;

	}

		cout << "Цепочка не найдена" << endl;
	}
