#include "pch.h"
#include <iostream>
#include <conio.h>
#include <math.h>
#include <iomanip>

using namespace std;


int main()
{
	setlocale(LC_ALL, "Russian");

	int N = 0;
	int **arr;

	while (N < 10) {
		cout << "Кол-во городов (N>=10) N=" << endl;
		cin >> N;
	}

	arr = new int *[N];

	for (int i = 0; i < N; i++) {
		arr[i] = new int[N];
	}

	for (int i = 0; i < N; i++) {
		for (int k = i + 1; k < N; k++) {
			arr[i][k] = rand() % 100 + 1;
		}
	}

	for (int i = 0; i < N; i++) {
		for (int k = i + 1; k < N; k++) {
			arr[k][i] = arr[i][k];
			arr[k][k] = 0;
			arr[i][i] = 0;
		}
	}
	

	cout << endl;

	for (int i = 0; i < N;i++) {
		cout<<" " << setw(5) << i;
	}
	cout << endl;

	cout << "____________________________________________________________________________________________________________________" << endl;
	for (int i = 0; i < N; i++) {
		cout << i << ")" << setw(5);
		for (int k = 0; k < N; k++) {
			cout << setw(5) << arr[i][k] << " ";
		}
		cout << endl;
		cout << endl;
	}

	cout << " " << endl;
	system("pause");
	return 0;
}
